# フレームバッファ先頭アドレスをRAXへ格納
00: 48 8b 06	# (RSI) -> RAX

# 画面の幅をEBXへ、高さをECXへ格納
03: 8b 5e 10	# 0x10(RSI) -> EBX
06: 8b 4e 14	# 0x14(RSI) -> ECX

# 無限Halt
09: f4		# Halt
0a: eb fd	# Jump -1
