# 1. ステータスレジスタ値を取得
00: e4 64	# (IO Addr 0x64) -> AL

# 2. キー入力値が無ければ先頭へ戻る
# (pauseしながら待つ)
02: 24 01	# AL and 0x01 -> AL
04: 75 04	# Jump(NotEqual) +6 (入力が有れば進む)
06: f3 90	# Pause
08: eb f6	# Jump -0x08 (入力が無ければ戻る)

# 3. データレジスタからキー入力値を取得
0a: e4 60	# (IO Addr 0x60) -> AL

# 4. キー入力値がBreakならスキャンコードを抽出し、Makeなら先頭へ戻る
0c: 2c 80	# AL - 0x80 -> AL
0e: 72 f0	# Jump(Below) -0x0e (Makeなら先頭へ戻る)

# 5. スキャンコードをレジスタCLへコピー
10: 88 c1	# AL -> CL

# 6. 先頭へ戻る
12: eb ec	# Jump -0x12
