# ステータスレジスタ値を取得
00: 66 ba 64 00	# 0x0064 -> dx
04: ec		# IO Addr(dx) -> al

# OBF(bit0)がセットされるまでpauseしながら待つ
05: f3 90	# pause
07: 24 01	# al & 0x01 -> al
09: 74 f5	# je -9(0x00)

# キー入力取得
0b: 66 ba 60 00	# 0x0060 -> dx
0f: ec		# IO Addr(dx) -> al

# breakであるか確認
10: 2c 80	# al - 0x80 -> al
12: 72 ec	# jb -18(0x00) (breakフラグが無いなら先頭へ戻る)

# breakのキー入力ならclレジスタへ格納
14: 88 c1	# al -> cl

# 先頭へ戻る
16: eb e8	# jmp -22(0x00)
