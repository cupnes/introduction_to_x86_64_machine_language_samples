# 1. ステータスレジスタ値を取得
00: e4 64	# IO Addr(0x64) -> al

# 2. キー入力値が無ければ先頭へ戻る
# (pauseしながら待つ)
02: 24 01	# al & 0x01 -> al
04: 75 04	# jne +6(0x0a) (入力が有れば進む)
06: f3 90	# pause
08: eb f6	# jmp -8(0x00) (入力が無ければ戻る)

# 3. データレジスタからキー入力値を取得
0a: e4 60	# IO Addr(0x60) -> al

# 4. キー入力値がBreakならスキャンコードを抽出し、Makeなら先頭へ戻る
0c: 2c 80	# al - 0x80 -> al
0e: 72 f0	# jb -14(0x00) (Makeなら先頭へ戻る)

# 5. スキャンコードをレジスタCLへコピー
10: 88 c1	# al -> cl

# 6. 先頭へ戻る
12: eb ec	# jmp -18(0x00)
