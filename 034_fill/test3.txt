# フレームバッファの先頭アドレスをRAXへコピー
48 8b 06	# (rsi) -> rax

# 幅をEBXへ、高さをECXへコピー
8b 5e 10	# 0x10(rsi) -> ebx
8b 4e 14	# 0x14(rsi) -> ecx

# cmpテスト
ba 01 53 07 00	# 0x75301 -> edx
89 ce		# ecx -> esi
0f af f3	# ebx * esi -> esi (0x75300)
39 d6		# ? esi - edx

# 無限ループhalt
f4		# hlt
eb fd		# jmp -1

# (qemu) info registers
# RAX=00000000c0000000 RBX=0000000000000320 RCX=0000000000000258 RDX=0000000000075301
# RSI=0000000000075300 RDI=00000000bfe9b018 RBP=00000000bff0eb10 RSP=0000000000210000
# R8 =00000000bff0e9dc R9 =0000000000000078 R10=00000000bfe6e080 R11=00000000e7dc4657
# R12=00000000bef69540 R13=00000000bef69548 R14=00000000bff24b60 R15=00000000bef6a018
# RIP=0000000000110016 RFL=00000097 [--S-APC] CPL=0 II=0 A20=1 SMM=0 HLT=1

# ESI == EDX の時、RFL=00000046 [---Z-P-]
# 変化があったビットは
# [セット] C(bit0): キャリーフラグ。最後の算術演算操作で加算においてレジスタの大きさを越えてビットのキャリー（桁上がり）かボロー（桁借り）をした場合にセットされる。
# [セット] AF(bit4): 調整フラグ。2進化10進 (BCD) 演算の算術演算でキャリーまたはボローが生じたらセットされる。
# [リセット] ZF(bit6): ゼロフラグ。操作の結果がゼロ (0) になった場合にセットされる。
# [セット] SF(bit7): 符号フラグ。操作の結果が負となった場合にセットされる。
