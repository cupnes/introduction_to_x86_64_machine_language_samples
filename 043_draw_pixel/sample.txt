# フレームバッファの先頭アドレスをRAXへコピー
48 8b 06	# (rsi) -> rax

# 幅をEBXへ、高さをECXへコピー
8b 5e 10	# 0x10(rsi) -> ebx
8b 4e 14	# 0x14(rsi) -> ecx

# 画面左上(0,0)を緑(0x00,0xff,0x00,0x00)で塗る
c7 00 00 ff 00 00	# (B,G,R,RSV)=(00,ff,00,00) -> (X,Y)=(0,0)

# 無限ループhalt
f4		# hlt
eb fd		# jmp -1
