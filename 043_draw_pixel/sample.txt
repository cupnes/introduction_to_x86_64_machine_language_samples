# フレームバッファ先頭アドレスをRAXへ
# 画面の幅をEBXへ、高さをECXへ格納
00: 48 8b 06		# (RSI) -> RAX
03: 8b 5e 10		# 0x10(RSI) -> EBX
06: 8b 4e 14		# 0x14(RSI) -> ECX

# フレームバッファ先頭ピクセル(画面左上)を緑で塗る
09: c7 00 00 ff 00 00	# (B,G,R,RSV)=(00,ff,00,00) -> (RAX)

# 無限Halt
0f: f4			# hlt
10: eb fd		# Jump -1

# [実行結果の例]
# (qemu) x/4bx 0xc0000000
# 00000000c0000000: 0x00 0xff 0x00 0x00
